from cryptography.fernet import Fernet
from cryptography.hazmat.primitives import hashes
from cryptography.hazmat.primitives.kdf.pbkdf2 import PBKDF2HMAC
import base64
import os

def derive_key(aes_secret: str) -> bytes:
    """Derive 32-byte Fernet key from AES_SECRET_KEY."""
    password = aes_secret.encode()
    salt = b'soccho_salt_2024'  # Fixed salt for determinism
    kdf = PBKDF2HMAC(
        algorithm=hashes.SHA256(),
        length=32,
        salt=salt,
        iterations=100000,
    )
    key = base64.urlsafe_b64encode(kdf.derive(password))
    return key

def encrypt_field(value: str, aes_secret: str) -> str:
    """AES-256 encrypt string field."""
    f = Fernet(derive_key(aes_secret))
    return f.encrypt(value.encode()).decode()

def decrypt_field(encrypted: str, aes_secret: str) -> str:
    """Decrypt."""
    f = Fernet(derive_key(aes_secret))
    return f.decrypt(encrypted.encode()).decode()
